Procedimiento saluda (Texto t1, Texto t2) {
   Escribir("Saludo de Despedida!");

   Para(0,3){
      Escribir(t1);
   }
   Escribir(t2);
   Escribir("Ing Christian :)");
}

Procedimiento cicloEscritura (Texto texto, Entero cant){
   Entero cont;
   cont = 0;
   Escribir("Inicio de ciclo");

   Mientras(cont<cant){
      Escribir(texto);
      cont = cont+1;
   }
   Escribir("Fin de ciclo");
}

Funcion Flotante diametro (Flotante radio){
   Flotante diam;
   diam = 2*radio;

   Si(diam > 0){
      Si(diam < 100){
         Escribir("Diametro calculado! Entre 0 y 100");
      } Sino{
         Escribir("Diametro calculado! Mayor que 0");
      }
   } Sino{
      Escribir("Diametro no puede ser menor que 0");
   }
   Retorna diam;
}

Funcion Entero suma (Entero num1, Entero num2){
   Entero resultado;
   resultado = num1+num2;

   Escribir("Suma realizada!");
   Retorna resultado;
}
